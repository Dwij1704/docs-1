- Completed the previous tutorials
- xpander.ai SDK installed

## Web Interface

```typescript
// Add web interface
const webUI = await agent.addInterface('web', {
  port: 3000,
  theme: 'light',
  features: ['chat', 'files']
});

// Configure UI components
webUI.addComponent({
  type: 'chat',
  position: 'right',
  width: '300px'
});

// Start the server
await webUI.start();
```

## Voice Interface

```typescript
// Add voice interface
const voice = await agent.addInterface('voice', {
  language: 'en-US',
  voice: 'neural-1'
});

// Handle voice input
voice.on('speech', async (audio) => {
  const text = await voice.transcribe(audio);
  const response = await agent.process(text);
  await voice.speak(response);
});

// Start listening
await voice.start();
```

## Slack Integration

```typescript
// Add Slack interface
const slack = await agent.addInterface('slack', {
  token: process.env.SLACK_TOKEN,
  channels: ['general']
});

// Handle messages
slack.on('message', async (event) => {
  const response = await agent.process(event.text);
  await slack.reply(event.channel, response);
});

// Add interactive components
slack.addComponent({
  type: 'button',
  text: 'Help',
  action: async (interaction) => {
    await slack.sendMessage(interaction.channel, 'How can I help?');
  }
});
```

## Complete Example

```typescript
import { XpanderClient } from '@xpander/sdk';

async function main() {
  const client = new XpanderClient({
    apiKey: 'your-api-key'
  });

  // Create agent
  const agent = await client.createAgent({
    name: 'MultiInterfaceAgent',
    description: 'An agent with multiple interfaces',
    model: 'gpt-4'
  });

  // Add web interface
  const webUI = await agent.addInterface('web', {
    port: 3000,
    theme: 'light'
  });

  // Add voice interface
  const voice = await agent.addInterface('voice', {
    language: 'en-US'
  });

  // Add Slack interface
  const slack = await agent.addInterface('slack', {
    token: process.env.SLACK_TOKEN
  });

  // Start all interfaces
  await Promise.all([
    webUI.start(),
    voice.start(),
    slack.connect()
  ]);

  console.log('All interfaces are running!');
}

main().catch(console.error);
```

## Interface Events

```typescript
// Listen for events across all interfaces
agent.on('message', async (event) => {
  console.log(`Message from ${event.interface}: ${event.text}`);
});

agent.on('error', (error) => {
  console.error(`Interface error: ${error.message}`);
});

agent.on('connection', (event) => {
  console.log(`New connection on ${event.interface}`);
});
```
