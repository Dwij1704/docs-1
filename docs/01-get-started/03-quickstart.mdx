## Create Your First Agent

```typescript
// Create a simple agent
const agent = await client.createAgent({
  name: 'MyFirstAgent',
  description: 'A simple assistant agent',
  model: 'gpt-4'
});

// Add basic capabilities
await agent.addCapability('conversation');
await agent.addCapability('tools');
```

## Make Your First Call

```typescript
// Send a message to your agent
const response = await agent.chat({
  message: 'Hello! Can you help me with a task?'
});

console.log(response.content);
```

<CardGroup cols={2}>

    await agent.addTool({
      name: 'calculator',
      description: 'Performs basic math operations',
      function: async (input) => {
        const { operation, numbers } = input;

            return numbers.reduce((a, b) => a + b, 0);
          case 'multiply':
            return numbers.reduce((a, b) => a * b, 1);

## Complete Example

```typescript
import { XpanderClient } from '@xpander/sdk';

async function main() {
  // Initialize client
  const client = new XpanderClient({
    apiKey: 'your-api-key'
  });

  // Create agent
  const agent = await client.createAgent({
    name: 'AssistantAgent',
    description: 'A helpful assistant',
    model: 'gpt-4'
  });

  // Add capabilities
  await agent.addCapability('conversation');
  await agent.addCapability('tools');

  // Add a tool
  await agent.addTool({
    name: 'calculator',
    description: 'Performs math operations',
    function: async (input) => {
      const { operation, numbers } = input;
      return numbers.reduce((a, b) => a + b, 0);
    }
  });

  // Start conversation
  const response = await agent.chat({
    message: 'Can you add 5, 10, and 15?'
  });

  console.log(response.content);
}

main().catch(console.error);
```
